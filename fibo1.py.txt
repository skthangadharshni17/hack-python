
def fibo(n):
   if n<=0:
      return "Invalid"
   elif n==1:
      return 0
   elif n==2:
      return 1
   else:
      return fibo(n-1)+fibo(n-2)

#main routine
n=int(input("Enter a number:"))
print("The fibonacci number is",end=" ")
for i in range(1,n+1):
   print(fibo(i),end=" ")
